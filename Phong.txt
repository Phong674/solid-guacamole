local ScreenGui = Instance.new("ScreenGui")

ScreenGui.Name = "TuanGayHub"

Frame.Parent = ScreenGui
Frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
Frame.Position = UDim2.new(0.4, 0, 0.4, 0)
Frame.Size = UDim2.new(0, 250, 0, 150)
Frame.Active = true
Frame.Draggable = true -- Có thể kéo menu

Title.Parent = Frame
Title.Size = UDim2.new(1, 0, 0, 40)
Title.Text = "TUẤN GÂY HUB"
Title.TextColor3 = Color3.fromRGB(255, 255, 255)
Title.BackgroundColor3 = Color3.fromRGB(200, 0, 0)

Logo.Parent = Frame
Logo.Position = UDim2.new(0.35, 0, 0.3, 0)
Logo.Size = UDim2.new(0, 70, 0, 70)
Logo.Image = "rbxassetid://119935888860104" -- Logo của bạn
Logo.BackgroundTransparency = 1

FarmBtn.Parent = Frame
FarmBtn.Position = UDim2.new(0.1, 0, 0.75, 0)
FarmBtn.Size = UDim2.new(0.8, 0, 0, 30)
FarmBtn.Text = "BẬT AUTO FARM [LEVEL]"
FarmBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 0)

-- Chức năng Auto Farm đơn giản
FarmBtn.MouseButton1Click:Connect(function()
    _G.AutoFarm = not _G.AutoFarm
    FarmBtn.Text = _G.AutoFarm and "ĐANG FARM..." or "BẬT AUTO FARM [LEVEL]"
    
    spawn(function()
        while _G.AutoFarm do
            wait()
            pcall(function()
                for i,v in pairs(game.Workspace.Enemies:GetChildren()) do
                    if v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.HumanoidRootPart.CFrame * CFrame.new(0,0,5)
                        local tool = game.Players.LocalPlayer.Backpack:FindFirstChildOfClass("Tool") or game.Players.LocalPlayer.Character:FindFirstChildOfClass("Tool")
                        if tool then tool:Activate() end
                    end
                end
            end)
        end
    end)
end)
