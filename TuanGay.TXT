-- TUẤN GÂY HUB [PHIÊN BẢN DELTA]
if game.CoreGui:FindFirstChild("TuanGayHub") then
    game.CoreGui.TuanGayHub:Destroy()
end

local ScreenGui = Instance.new("ScreenGui", game.CoreGui)
ScreenGui.Name = "TuanGayHub"

local Main = Instance.new("Frame", ScreenGui)
Main.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
Main.Position = UDim2.new(0.35, 0, 0.35, 0)
Main.Size = UDim2.new(0, 250, 0, 180)
Main.Active = true
Main.Draggable = true

local Title = Instance.new("TextLabel", Main)
Title.Size = UDim2.new(1, 0, 0, 40)
Title.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
Title.Text = "TUẤN GÂY HUB [DELTA]"
Title.TextColor3 = Color3.new(1,1,1)
Title.TextSize = 18

local Logo = Instance.new("ImageLabel", Main)
Logo.Position = UDim2.new(0.35, 0, 0.25, 0)
Logo.Size = UDim2.new(0, 70, 0, 70)
Logo.Image = "rbxassetid://119935888860104"
Logo.BackgroundTransparency = 1

local FarmBtn = Instance.new("TextButton", Main)
FarmBtn.Position = UDim2.new(0.1, 0, 0.7, 0)
FarmBtn.Size = UDim2.new(0.8, 0, 0, 40)
FarmBtn.BackgroundColor3 = Color3.fromRGB(0, 180, 0)
FarmBtn.Text = "BẬT AUTO FARM"
FarmBtn.TextColor3 = Color3.new(1,1,1)

_G.Auto = false
FarmBtn.MouseButton1Click:Connect(function()
    _G.Auto = not _G.Auto
    FarmBtn.Text = _G.Auto and "ĐANG FARM [ON]" or "BẬT AUTO FARM"
    while _G.Auto do
        task.wait()
        pcall(function()
            for _,v in pairs(workspace.Enemies:GetChildren()) do
                if v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                    game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.HumanoidRootPart.CFrame * CFrame.new(0,0,5)
                    local tool = game.Players.LocalPlayer.Backpack:FindFirstChildOfClass("Tool") or game.Players.LocalPlayer.Character:FindFirstChildOfClass("Tool")
                    if tool then tool:Activate() end
                end
            end
        end)
    end
end
